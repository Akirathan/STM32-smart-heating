246 static double convert_positive_temperature(uint8_t lsb, uint8_t msb, resolution_t resolution)
247 {
248  // LSB
249  int lower_bound = -4;

251

254  lower_bound = -4;
255  break;
256  case RESOLUTION_11_BIT:
257  lower_bound = -3;

259  break;
260  case RESOLUTION_10_BIT:
261  lower_bound = -2;
262  lsb >>= 2;
263  break;
264  case RESOLUTION_9_BIT:
265  lower_bound = -1;
266  lsb >>= 3;
267  break;
268  }
269

271  for (int i = lower_bound; i <= upper_bound; i++) {
272  uint8_t last_bit = lsb & 0x01;
273  if (last_bit)
274  sum += pow(2, i);
275  lsb >>= 1;
276  }
277 
278  // MSB

280  uint8_t last_bit = msb & 0x01;
281  if (last_bit)
